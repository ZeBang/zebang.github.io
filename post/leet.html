Example 1:

xxxxxxxxxx
Input: nums = [3,2,3]
Output: 3
xxxxxxxxxx
class Solution:
    def majorityElement(self, nums):
        count = 0
        for num in nums:
            if count == 0: candidate = num
            count += (1 if num == candidate else -1)
        return candidate
