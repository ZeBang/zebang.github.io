1.7 曲面逼近
割面逼近不一定有效

% 画圆柱
[X Y Z] = cylinder(1, 50);
Z = Z * 2;
surf(X, Y, Z); hold on;
shading interp
alpha(0.25);
% 画割面
n = 5; % 将圆 n 等分
m = 6; % 将高 m 等分
for i = 1 : m+1
for j = 1 : n + 1
11
X(i,j) = cos((2 * j + mod(i,2)) * pi / n);
Y(i,j) = sin((2 * j + mod(i,2)) * pi / n);
Z(i,j) = 2 * (i-1) / m;
end
end
for i = 1 : m
for j = 1 : n
x = [X(i,j) X(i,j+1) X(i+1,j+mod(i,2))];
y = [Y(i,j) Y(i,j+1) Y(i+1,j+mod(i,2))];
z = [Z(i,j) Z(i,j+1) Z(i+1,j+mod(i,2))];
fill3(x, y, z, ’g’);
end
end
for i = 2 : m + 1
for j = 1 : n
x = [X(i,j) X(i,j+1) X(i-1,j+mod(i,2))];
y = [Y(i,j) Y(i,j+1) Y(i-1,j+mod(i,2))];
z = [Z(i,j) Z(i,j+1) Z(i-1,j+mod(i,2))];
fill3(x, y, z, ’y’);
end
